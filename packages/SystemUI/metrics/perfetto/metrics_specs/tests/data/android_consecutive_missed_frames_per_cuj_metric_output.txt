  specs {
    id: "max_consecutive_missed_frames"
    dimensions: "cuj_name"
    dimensions: "cuj_start"
    value: "missed_frames"
    query {
      sql {
        sql: "\n        INCLUDE PERFETTO MODULE android.cujs.sysui_cujs;\n        INCLUDE PERFETTO MODULE android.frames.jank_type;\n\n        WITH frames_during_cuj AS (\n          SELECT\n            fts.ts AS ts,\n            fts.jank_type AS jank_type,\n            cuj.cuj_name AS cuj_name,\n            cuj.ts AS cuj_start,\n            fts.name AS frame_number,\n            -- Create a simple 1 or 0 flag now to make later logic cleaner\n            android_is_missed_frame_type(fts.jank_type) AS is_jank,\n            android_is_sf_jank_type(fts.jank_type) AS is_sf_jank,\n            android_is_app_jank_type(fts.jank_type) AS is_app_jank\n          FROM actual_frame_timeline_slice AS fts\n          JOIN android_sysui_jank_cujs AS cuj\n            ON fts.upid = cuj.upid\n            AND fts.ts >= cuj.ts           -- Frame must start after CUJ starts\n            AND fts.ts < cuj.ts + cuj.dur  -- Frame must start before CUJ ends\n        ),\n\n        -- Solve the gaps-and-islands to find the consecutive runs\n        -- for this, we need to keep track of the previous lag(is_*_jank)\n        state_changes AS (\n          SELECT\n            *,\n            CASE\n              WHEN is_jank != LAG(is_jank)\n                OVER (PARTITION BY cuj_name, cuj_start ORDER BY ts) THEN 1 ELSE 0\n            END AS jank_state_changed,\n            CASE\n              WHEN is_sf_jank != LAG(is_sf_jank)\n                OVER (PARTITION BY cuj_name, cuj_start ORDER BY ts) THEN 1 ELSE 0\n            END AS sf_jank_state_changed,\n            CASE\n              WHEN is_app_jank != LAG(is_app_jank)\n                OVER (PARTITION BY cuj_name, cuj_start ORDER BY ts) THEN 1 ELSE 0\n            END AS app_jank_state_changed\n          FROM frames_during_cuj\n        ),\n\n        -- Create Streak IDs (ISOLATED PER CUJ)\n        streak_identification AS (\n          SELECT\n            *,\n            -- Create a running ID that is unique to THIS specific CUJ instance\n            SUM(jank_state_changed)\n              OVER (PARTITION BY cuj_name, cuj_start ORDER BY ts) AS streak_id,\n            SUM(sf_jank_state_changed)\n              OVER (PARTITION BY cuj_name, cuj_start ORDER BY ts) AS sf_streak_id,\n            SUM(app_jank_state_changed)\n              OVER (PARTITION BY cuj_name, cuj_start ORDER BY ts) AS app_streak_id\n          FROM state_changes\n        ),\n\n        -- Count the consecutive missed frames for each CUJ for each streak\n        all_streak_lengths AS (\n          SELECT\n            *,\n            COUNT(*) OVER (PARTITION BY cuj_name, cuj_start, streak_id) AS streak_length,\n            COUNT(*) OVER (PARTITION BY cuj_name, cuj_start, sf_streak_id) AS sf_streak_length,\n            COUNT(*) OVER (PARTITION BY cuj_name, cuj_start, app_streak_id) AS app_streak_length\n          FROM streak_identification\n        )\n\n        -- Filter out to choose the maximum consecutive missed frames\n        -- Collapse by CUJ and its start time and only select the MAX length\n        -- where the frames were actually janky\n        SELECT\n          cuj_name,\n          cuj_start,\n          -- If is_jank is 0, we treat the length as 0 so it doesn\'t become the MAX\n          CAST(MAX(CASE WHEN is_jank = 1 THEN streak_length ELSE 0 END) AS REAL)\n            AS missed_frames,\n          CAST(MAX(CASE WHEN is_sf_jank = 1 THEN sf_streak_length ELSE 0 END) AS REAL)\n            AS missed_sf_frames,\n          CAST(MAX(CASE WHEN is_app_jank = 1 THEN app_streak_length ELSE 0 END) AS REAL)\n            AS missed_app_frames\n        FROM all_streak_lengths\n        GROUP BY cuj_name, cuj_start;\n      "
      }
    }
    bundle_id: "max_consecutive"
